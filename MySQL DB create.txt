drop database crm;
create database crm;
use crm;
create table address(addressId int NOT NULL AUTO_INCREMENT,
					 addressName varchar(20),
                     address2Name varchar(20),
                     cityId int,
                     postalCode varChar(20),
                     phone varchar(20),
                     createDate varchar(40),
                     createdBy varchar(20),
                     lastUpdate varchar(40),
                     lastUpdateBy varchar(20),
					 PRIMARY KEY (addressId));
create table appointment(appointmentId int NOT NULL AUTO_INCREMENT,
						 customerId int,
                         userId int,
                         title varchar(20),
                         description varchar(20),
                         location varchar(20) ,
                         contact varchar(20),
                         type varchar(20),
                         url varchar(20),
                         start varchar(20),
                         end varchar(20),
						 createDate varchar(40),
						 createdBy varchar(20),
						 lastUpdate varchar(40),
						 lastUpdateBy varchar(20),
						 PRIMARY KEY (appointmentId));
create table city(cityId int NOT NULL AUTO_INCREMENT,
				  cityName varchar(20),
                  countryId int,
				  createDate varchar(40),
				  createdBy varchar(20),
				  lastUpdate varchar(40),
				  lastUpdateBy varchar(20),
				  PRIMARY KEY (cityId));
create table country(countryId int NOT NULL AUTO_INCREMENT,
					 countryName varChar(20),
                     createDate varchar(40),
                     createdBy varchar(20),
                     lastUpdate varchar(40),
                     lastUpdateBy varchar(20),
					 PRIMARY KEY (countryId));
create table customer(customerId int NOT NULL AUTO_INCREMENT,
					  customerName varchar(20),
                      addressId int,
                      active bool,
                      createDate varchar(40),
                      createdBy varchar(20),
                      lastUpdate varchar(40),
                      lastUpdateBy varchar(20),
					  PRIMARY KEY (customerId));
create table report( reportId int NOT NULL AUTO_INCREMENT,
					 reportName varchar(20),
				     PRIMARY KEY (reportId));
create table user(userId int NOT NULL AUTO_INCREMENT,
				  userName varchar(20),
				  password varchar(20),
                  active bool,
				  createDate varchar(40),
				  createdBy varchar(20),
				  lastUpdate varchar(40),
				  lastUpdateBy varchar(20),
				  PRIMARY KEY (userId));
